#!/bin/bash

slow_type() {
    text="$1"
    delay="${2:-0.03}"
    for (( i=0; i<${#text}; i++ )); do
        echo -ne "${text:$i:1}"
        sleep $delay
    done
    echo
}

clear
RED='\033[1;31m'
YELLOW='\033[1;33m'
GREEN='\033[1;32m'
NC='\033[0m'

echo -e "${YELLOW}"
slow_type "═══════════════════════════════════════════════"
slow_type "          WELCOME TO VAMPIRE-HUB-X [ADMIN MODE]"
slow_type "═══════════════════════════════════════════════"
echo -e "${RED}"
slow_type "[!] WARNING: This tool grants elevated powers..."
slow_type "    Including but not limited to:"
slow_type "    - Real-Time DDoS Launching"
slow_type "    - Deep Info-Gathering on Targets"
slow_type "    - IP Analysis and Tracking"
slow_type "    - Cloaked Admin Features (Encrypted)"
echo ""
echo -e "${YELLOW}"
slow_type "[!] This mode is reserved for Vampire Squad Authorized Hackers Only."
slow_type "[!] Unauthorized access will be tracked and countered."
slow_type "[!] All actions are logged (Locally only) for security."
echo ""
echo -e "${GREEN}"
slow_type "Proceed only if you understand the risks..."
slow_type "And you are authorized by Muhammad Shourov (VAMPIRE)"
slow_type "Respect the Code. Use it Ethically. Fear the Vampire."

echo -e "${NC}"
sleep 2
